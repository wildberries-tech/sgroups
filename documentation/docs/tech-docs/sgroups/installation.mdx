      git checkout tags/v${VERSION}

    ```bash title='Configuration file setup'
    export VERSION=$(cat VERSION | cut -c2-)
    cat <<EOF > config-server.yaml
    ---
    logger:
        # log level
        level: INFO

    metrics:
        # enable api metrics
        enable: true

        # enables|disables health check handler
        enable: true

        # graceful shutdown period
        graceful-shutdown: 30s
    EOF
    ```

    ```docker
    FROM golang:1.20-alpine3.17 as builder
    WORKDIR /src
    COPY . .
    RUN apk upgrade && \
        apk add make && \
        make sg-service platform=linux/amd64

    FROM alpine:3.17
    RUN addgroup -g 101 app && \
        adduser -H -u 101 -G app -s /bin/sh -D app && \
        apk update && apk upgrade
    WORKDIR /app/
    RUN mkdir -p /app/bin
    COPY --from=builder --chown=app:app /src/bin/ /app/bin/
    USER app
    CMD ["/app/bin/sgroups"]
    ```

    ```bash
    docker build --tag sgroups:v$(cat VERSION) .
    ```

    ```
    docker run \
    -d \
    -v config-server.yaml:/opt/swarm/etc/sgroups/config-server.yaml \
    --name hbf-server \
    --entrypoint "/app/bin/sgroups" \
    sgroups:v$(cat VERSION) -config /opt/swarm/etc/sgroups/config-server.yaml

    ```

  Для установки deb-пакета необходимо воспользоваться скриптом ниже с указанием необходимой версии в строке `export

```bash title='bash'
export VERSION="1.9.1"
export PACKAGE_TYPE=deb
export URL=https://github.com/H-BF/sgroups/releases/download
export RELEASE=v$VERSION/sgroups-$VERSION-any.$PACKAGE_TYPE

sudo wget -O /tmp/sgroups $URL/$RELEASE
sudo dpkg -i /tmp/sgroups
systemctl enable sgroups
systemctl start sgroups
```

<TabItem value='source'>

    ```bash title='bash'
    ## INSTALL SERVER
    export VERSION=v1.9.1
    git clone https://github.com/H-BF/sgroups
    cd sgroups
    git checkout tags/${VERSION}
    make sg-service platform=linux/amd64
    mkdir -p /opt/swarm/sbin
    mkdir -p /opt/swarm/etc/sgroups
    cp bin/sgroups /opt/swarm/sbin/sgroups

    cat <<EOF > /opt/swarm/etc/sgroups/config-server.yaml
    ---
    logger:
        # log level
        level: INFO

    metrics:
        # enable api metrics
        enable: true

    healthcheck:
        # enables|disables health check handler
        enable: true

    server:
        # server endpoint
        endpoint: tcp://0.0.0.0:9650
        # graceful shutdown period
        graceful-shutdown: 30s
    EOF

    cat <<EOF > /etc/systemd/system/hbf-server.service
    [Unit]
    Description=sgroups
    Documentation=TODO:
    After=network.target

    [Service]
    ExecStart=/opt/swarm/sbin/sgroups --config=/opt/swarm/etc/sgroups/config-server.yaml
    Restart=always
    RestartSec=5
    Delegate=yes
    KillMode=process
    OOMScoreAdjust=-999
    LimitNOFILE=1048576
    LimitNPROC=infinity
    LimitCORE=infinity

    [Install]
    WantedBy=multi-user.target
    EOF

    systemctl enable hbf-server.service
    systemctl start hbf-server.service
    ```

## Настройка

Для настройки серверной части требуется использовать конфигурационный файл, который содержит поля, позволяющие настраивать параметры в соответствии с потребностями пользователей.

```yaml title='YAML'
---
logger:
  # log level
  level: INFO

metrics:
  # enable api metrics
  enable: true

healthcheck:
  # enables|disables health check handler
  enable: true

server:
  # server endpoint
  endpoint: tcp://0.0.0.0:9650
  # graceful shutdown period
  graceful-shutdown: 30s
```
